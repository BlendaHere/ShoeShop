[HttpPut]
public JsonResult Put(LoginRole loginRole)
{
    string query = @"
        UPDATE LoginRole
        SET Username = @Username,
            Role = @Role
        WHERE LoginId = @LoginId";

    string sqlDataSource = _configuration.GetConnectionString("EmployeeAppCon");
    using (SqlConnection myCon = new SqlConnection(sqlDataSource))
    {
        myCon.Open();
        using (SqlCommand myCommand = new SqlCommand(query, myCon))
        {
            myCommand.Parameters.AddWithValue("@LoginId", loginRole.LoginId);
            myCommand.Parameters.AddWithValue("@Username", loginRole.Username);
            myCommand.Parameters.AddWithValue("@Role", loginRole.Role);
            myCommand.ExecuteNonQuery();
        }
        myCon.Close();
    }

    return new JsonResult("Updated Successfully");
}
