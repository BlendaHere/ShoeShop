[HttpPut]
public JsonResult Put(OrderForRepairOrCleaning order)
{
    string query = @"
        UPDATE OrderForRepairOrCleaning
        SET CustomerName = @CustomerName,
            ShoeBrand = @ShoeBrand,
            ShoeColor = @ShoeColor,
            RepairOrCleaningType = @RepairOrCleaningType,
            OrderDate = @OrderDate,
            IsCompleted = @IsCompleted
        WHERE OrderId = @OrderId";

    string sqlDataSource = _configuration.GetConnectionString("EmployeeAppCon");
    using (SqlConnection myCon = new SqlConnection(sqlDataSource))
    {
        myCon.Open();
        using (SqlCommand myCommand = new SqlCommand(query, myCon))
        {
            myCommand.Parameters.AddWithValue("@OrderId", order.OrderId);
            myCommand.Parameters.AddWithValue("@CustomerName", order.CustomerName);
            myCommand.Parameters.AddWithValue("@ShoeBrand", order.ShoeBrand);
            myCommand.Parameters.AddWithValue("@ShoeColor", order.ShoeColor);
            myCommand.Parameters.AddWithValue("@RepairOrCleaningType", order.RepairOrCleaningType);
            myCommand.Parameters.AddWithValue("@OrderDate", order.OrderDate);
            myCommand.Parameters.AddWithValue("@IsCompleted", order.IsCompleted);

            myCommand.ExecuteNonQuery();
        }
    }

    return new JsonResult("Updated Successfully");
}
