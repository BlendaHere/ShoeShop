[HttpPost]
public JsonResult Insert(Offer offer)
{
    string query = @"
        INSERT INTO Offers (OfferName, Discount, StartDate, EndDate)
        VALUES (@OfferName, @Discount, @StartDate, @EndDate)";

    string sqlDataSource = _configuration.GetConnectionString("EmployeeAppCon");
    using (SqlConnection myCon = new SqlConnection(sqlDataSource))
    {
        myCon.Open();
        using (SqlCommand myCommand = new SqlCommand(query, myCon))
        {
            myCommand.Parameters.AddWithValue("@OfferName", offer.OfferName);
            myCommand.Parameters.AddWithValue("@Discount", offer.Discount);
            myCommand.Parameters.AddWithValue("@StartDate", offer.StartDate);
            myCommand.Parameters.AddWithValue("@EndDate", offer.EndDate);
            myCommand.ExecuteNonQuery();
        }
        myCon.Close();
    }

    return new JsonResult("Added Successfully");
}
